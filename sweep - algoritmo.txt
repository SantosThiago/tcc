entrada:número de veículos,capacidade,grafo
saída:grafo

cluster atual <- 1
angulos polares <- {}
nós <- {}
g <- {}
x_nó <- pos_x(nó) - pos_x(depósito)
y_nó <- pos_y(nó) - pos_y(depósito)

faça v em V:
	p <- x_nó/raiz((x_nó)^2+(y_nó)^2)
	ang <- arccos(p)*180/pi

	caso o y_nó<0:
		ang <-ang*(-1)
	
	
	adiciona(ang,angulos_polares)

ordena_crescente(angulos polares)

faça a em angulos polares:
	
	caso a capacidade>=demanda(v):
		capacidade=capacidade-demanda(v)

	caso não:
		
		adiciona(nós,g)
		nós <- {}
		reinicia(capacidade)
		cluster atual <- cluster atual + 1
		capacidade = capacidade - demanda(v)
	
	adiciona(v,nós)

adiciona(nós,g)
ordena_por_tempo(g)
caixeiro_viajante(g)

retorna g
